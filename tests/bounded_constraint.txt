# Test: Bounded integer (percentage 0-100)
# Feature: Value must be in [0, 100]
[test::Percentage]
description=int; bounded<0,100>, <=>

# Test: Bounded floating point (temperature)
# Feature: Absolute zero to sun's core
[physics::Temperature]
description=double; bounded<-273.15,1e7>, +, -, *, /, c++20

# Test: Bounded with checked arithmetic
# Feature: Percentage with overflow checking
[test::BoundedChecked]
description=std::uint8_t; bounded<0,100>, +, -, checked, #<cstdint>

# Test: Narrow bounds (edge case)
# Feature: Only one valid value
[test::FortyTwo]
description=int; bounded<42,42>, ==

# Test: String bounds
[test::BoundedString]
description=std::string; bounded<"A", "AAAA">, <=>
